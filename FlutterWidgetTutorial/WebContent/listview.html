Dynamic ListView Widget 

import 'package:flutter/material.dart';

class ListViewTutorialTutorial extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("ListView Tutorial"),
      ),
      body: Container(
        color: Colors.yellow,
        child: ListView.builder(
          itemCount: 10,
          itemBuilder: (context, position) {
            return _listItem();
          },
        ),
      ),
    );
  }

  _listItem() {
    return Card(
      elevation: 10,
      shadowColor: Colors.red,
      child: Text(
        "A",
        style: TextStyle(
          fontWeight: FontWeight.bold,
          fontSize: 25,
          fontFamily: "serif",
        ),
      ),
    );
  }
}
Dynamic ListView Widget with Separator 

import 'package:flutter/material.dart';

class RowTutorial extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("Row Tutorial"),
      ),
      body: Container(
        color: Colors.yellow,
        child: ListView.separated(
          itemCount: 10,
          itemBuilder: (context, position) {
            return _listItem();
          },
          separatorBuilder: (context, index) {
            return _listSeparated();
          },
        ),
      ),
    );
  }

  _listItem() {
    return Card(
      elevation: 10,
      shadowColor: Colors.red,
      child: Text(
        "A",
        style: TextStyle(
          fontWeight: FontWeight.bold,
          fontSize: 25,
          fontFamily: "serif",
        ),
      ),
    );
  }

  _listSeparated() {
    return Padding(
      padding: const EdgeInsets.all(8.0),
      child: Container(
        width: 100,
        height: 30,
        color: Colors.white,
      ),
    );
  }
}
